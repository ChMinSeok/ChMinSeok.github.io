## 연결리스트로 구현한 스택
```c++
#include <bits/stdc++.h>
#define ll long long
using namespace std;

class Node {
public:
    Node* prev;
    Node* next;
    int data;

    Node(int data = 0) : prev(nullptr), next(nullptr), data(data) {}
};
class Lstack
{
public:
    Node* head;
    Node* cur;
    int sizeN = 0;

    Lstack()
    {
        head = new Node();
        cur = head;
    }

    void push(int data)
    {
        sizeN++;
        Node* newNode = new Node(data);
        if (head->next == NULL)
        {
            head->next = newNode;
            newNode->prev = head;
            cur = newNode;

            return;
        }
        cur->next = newNode;
        newNode->prev = cur;
        cur = newNode;
        return;
    }

    void pop()
    {
        if (sizeN == 0)
        {
            cout << "-1\n";
            return;
        }
        else
            cout << cur->data << "\n";
        sizeN--;
        Node* temp = cur;
        cur = cur->prev;
        delete temp;
        return;
           
    }
    void top()
    {
        if (sizeN == 0)
        {
            cout << "-1\n";
            return;
        }
           
        cout << cur->data << "\n";
    }
    void size()
    {
        cout << sizeN << "\n";
    }
    void empty()
    {
        if (sizeN == 0)
            cout << "1\n";
        else
            cout << "0\n";
    }

};
int main(void) {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int n; cin >> n;
    Lstack st;
    for (int i = 0; i < n; i++)
    {
        string s; cin >> s;
        if (s == "push")
        {
            int tmp; cin >> tmp;
            st.push(tmp);
        }
        if (s == "top")
        {
            st.top();
        }
        if (s == "size")
        {
            st.size();
        }
        if (s == "pop")
        {
            st.pop();
        }
        if (s == "empty")
            st.empty();
    }
    return 0;
}
```
